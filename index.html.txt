    (function() {

        heart.style.animationDelay = Math.random() * 15 + 's';
        heart.style.fontSize = (Math.random() * 2 + 1.2) + 'rem';
        heart.style.opacity = 0.2;
        bgHearts.appendChild(heart);
      }

      // ---- mischievous NO button ----
      const noBtn = document.getElementById('noBtn');
      const yesBtn = document.getElementById('yesBtn');
      const proposalPage = document.getElementById('proposalPage');
      const secondPage = document.getElementById('secondPage');
      const container = document.getElementById('appContainer');

      // make the NO button run away
      if (noBtn) {
        noBtn.addEventListener('mouseover', function(e) {
          // playful evasion - move to random nearby position, but not too far
          const x = Math.random() * 200 - 100; // -100 to 100
          const y = Math.random() * 200 - 100;
          // but keep it inside the viewport roughly
          const currentLeft = noBtn.offsetLeft;
          const currentTop = noBtn.offsetTop;
          // apply transform
          noBtn.style.transform = `translate(${x}px, ${y}px)`;
          noBtn.style.transition = 'transform 0.2s ease';
        });

        // also evade on touch for mobile
        noBtn.addEventListener('touchstart', function(e) {
          e.preventDefault();
          const x = Math.random() * 200 - 100;
          const y = Math.random() * 200 - 100;
          noBtn.style.transform = `translate(${x}px, ${y}px)`;
          noBtn.style.transition = 'transform 0.2s ease';
        });
      }

      // ---- YES button click ----
      yesBtn.addEventListener('click', function(e) {
        // 1. heart explosion
        for (let i = 0; i < 28; i++) {
          createBurstHeart();
        }

        // 2. hide proposal card with fade, show second page
        proposalPage.style.opacity = '0';
        setTimeout(() => {
          proposalPage.style.display = 'none';
          secondPage.classList.add('show');
          // smooth scroll to top
          window.scrollTo({ top: 0, behavior: 'smooth' });
        }, 500);

        // 3. start relationship counter (update every second)
        startCounter();
      });

      // burst heart factory
      function createBurstHeart() {
        const burstContainer = document.getElementById('heartBurst');
        const heartEl = document.createElement('div');
        heartEl.className = 'burst-heart';
        heartEl.innerText = ['‚ù§Ô∏è','üß°','üíñ','üíó','üíò','üíù','üå∏','üåπ','üíï'][Math.floor(Math.random()*9)];
        // random start position around center
        heartEl.style.left = (50 + (Math.random()*40 - 20)) + '%';
        heartEl.style.top = (50 + (Math.random()*40 - 20)) + '%';
        // random direction
        const dx = (Math.random()*400 - 200) + 'px';
        const dy = (Math.random()*400 - 200) + 'px';
        heartEl.style.setProperty('--dx', dx);
        heartEl.style.setProperty('--dy', dy);
        burstContainer.appendChild(heartEl);
        setTimeout(() => heartEl.remove(), 1500);
      }

      // counter start from March 4, 2024
      function startCounter() {
        const startDate = new Date('2024-03-04T00:00:00'); // March 4, 2024

        function updateCounter() {
          const now = new Date();
          let diff = now - startDate;
          if (diff < 0) diff = 0;

          const seconds = Math.floor(diff / 1000) % 60;
          const minutes = Math.floor(diff / (1000 * 60)) % 60;
          const hours = Math.floor(diff / (1000 * 60 * 60)) % 24;
          const days = Math.floor(diff / (1000 * 60 * 60 * 24)) % 30; // approx month days
          const months = Math.floor(diff / (1000 * 60 * 60 * 24 * 30)) % 12;
          const years = Math.floor(diff / (1000 * 60 * 60 * 24 * 365));

          document.getElementById('years').innerText = years;
          document.getElementById('months').innerText = months;
          document.getElementById('days').innerText = days;
          document.getElementById('hours').innerText = hours;
          document.getElementById('minutes').innerText = minutes;
          document.getElementById('seconds').innerText = seconds;
        }

        updateCounter();
        setInterval(updateCounter, 1000);
      }

      // extra: if someone tries to click NO anyway, we can also move it on click (but it's already evading)
      noBtn.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();

      });

      // ensure NO button resets position not to go off screen too far (but it's fine)
      // also make it harder on mobile by moving on touch
    })();
